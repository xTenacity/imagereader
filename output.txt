            movey = 14;

            movex = 3;
        } else if (part.equals("4")) {
            movex = 2;
            movey = 5;
        } else if (part.equals("5")) {
            movex = 7;
            movey = 1;
        } else if (part.equals("6")) {
            movex = 15;
            movey = 19;
        } else if (part.equals("7")) {

            movey = 3;

            movex = 14;
            movey = 19;

            movey = 21;

            movex = 13;
            movey = 19;
        } else if (part.equals("11")) {
            movex = 14;
            movey = 22;

            movex = 5;
            movey = 8;
