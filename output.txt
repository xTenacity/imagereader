import org.code.neighborhood.*;
public class ImagePainter extends PainterPlus {//initialize the ImagePainter class
    public void drawImage(String color0, String color1, String color2) {
        drawColor0(color0);
        drawColor1(color1);
        drawColor2(color2);
    }

    public void drawColor0(String color) {
        //no moveToStart function because the initial position is at 0, 0
        paintLine(color, 12);
        nextLayer();
        paintLine(color, 12);
        nextLayer();
        paintLine(color, 12);
        nextLayer();
        paintLine(color, 3);
        move(7);
        paintLine(color, 3);
        nextLayer();
        paintLine(color, 2);
        move(3);
        paintLine(color, 4);
        move(3);
        paintLine(color, 2);
        nextLayer();
        paintLine(color, 2);
        move(2);
        paintLine(color, 2);
        move(3);
        paintLine(color, 2);
        move(2);
        paintLine(color, 2);
        nextLayer();
        paintLine(color, 2);
        move(2);
        paintLine(color, 1);
        move(5);
        paintLine(color, 1);
        move(2);
        paintLine(color, 2);
        nextLayer();
        paintLine(color, 2);
        move(2);
        paintLine(color, 6);
        move(2);
        paintLine(color, 2);
        nextLayer();
        paintLine(color, 2);
        move(9);
        paintLine(color, 2);
        nextLayer();
        paintLine(color, 2);
        move(9);
        paintLine(color, 2);
        nextLayer();
        paintLine(color, 2);
        move(9);
        paintLine(color, 2);
        nextLayer();
        paintLine(color, 2);
        move(9);
        paintLine(color, 2);
    home();
    }
    public void drawColor1(String color) {
        moveToStart("1");
        move(2);
        paintLine(color, 2);
        move(1);
        nextLayer();
        move(4);
    home();
    }
    public void drawColor2(String color) {
        moveToStart("2");
        move(2);
        paintLine(color, 6);
        move(1);
        nextLayer();
        move(3);
        paintLine(color, 2);
        move(4);
        nextLayer();
        paintLine(color, 1);
        move(7);
        paintLine(color, 1);
        nextLayer();
        move(3);
        paintLine(color, 1);
        move(5);
        nextLayer();
        paintLine(color, 1);
        move(7);
        paintLine(color, 1);
        nextLayer();
        move(3);
        paintLine(color, 6);
        nextLayer();
        paintLine(color, 8);
        nextLayer();
        move(3);
        paintLine(color, 6);
        nextLayer();
        paintLine(color, 8);
    home();
    }
    public void moveToStart(String part) {
        if (part.equals("1")) {
            move(3);
            turnRight();
            move(5);
            turnLeft();
        }
 else if (part.equals("2")) {
            move(1);
            turnRight();
            move(3);
            turnLeft();
        }

    }
}
