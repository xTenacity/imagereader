// Put this in the ImagePainter file
import org.code.neighborhood.*;
public class ImagePainter extends PainterPlus {// initialize the ImagePainter class
    public void drawImage(String color0, String color1, String color2) {
        drawColor0(color0);
        drawColor1(color1);
        drawColor2(color2);
    }

    public void drawColor0(String color) {
        // no moveToStart function because the initial position is at 0, 0
            paintLine(color, 12);
            nextLayer();                        //hi
            paintLine(color, 12);
            nextLayer();                        //hi
            paintLine(color, 12);
            nextLayer();                        //hi
            paintLine(color, 3);
            move(5);
            paintLine(color, 4);
            nextLayer();                        //hi
            paintLine(color, 2);
            move(1);
            paintLine(color, 5);
            move(1);
            paintLine(color, 3);
            nextLayer();                        //hi
            paintLine(color, 2);
            move(0);
            paintLine(color, 3);
            move(1);
            paintLine(color, 3);
            move(0);
            paintLine(color, 3);
            nextLayer();                        //hi
            paintLine(color, 2);
            move(0);
            paintLine(color, 2);
            move(3);
            paintLine(color, 2);
            move(0);
            paintLine(color, 3);
            nextLayer();                        //hi
            paintLine(color, 2);
            move(0);
            paintLine(color, 7);
            move(0);
            paintLine(color, 3);
            nextLayer();                        //hi
            paintLine(color, 2);
            move(7);
            paintLine(color, 3);
            nextLayer();                        //hi
            paintLine(color, 2);
            move(7);
            paintLine(color, 3);
            nextLayer();                        //hi
            paintLine(color, 2);
            move(7);
            paintLine(color, 3);
            nextLayer();                        //hi
            paintLine(color, 2);
            move(7);
            paintLine(color, 3);
        home();
    }
    public void drawColor1(String color) {
        moveToStart("1");
            move(0);
            paintLine(color, 3);
            move(1);
            nextLayer();                        //hi
            move(4);
        home();
    }
    public void drawColor2(String color) {
        moveToStart("2");
            move(0);
            paintLine(color, 7);
            move(1);
            nextLayer();                        //hi
            move(1);
            paintLine(color, 3);
            move(4);
            nextLayer();                        //hi
            paintLine(color, 1);
            move(5);
            paintLine(color, 2);
            nextLayer();                        //hi
            move(1);
            paintLine(color, 2);
            move(5);
            nextLayer();                        //hi
            paintLine(color, 1);
            move(5);
            paintLine(color, 2);
            nextLayer();                        //hi
            move(1);
            paintLine(color, 7);
            nextLayer();                        //hi
            paintLine(color, 8);
            nextLayer();                        //hi
            move(1);
            paintLine(color, 7);
            nextLayer();                        //hi
            paintLine(color, 8);
        home();
    }
    public void moveToStart(String part) {
        int movex = 0, movey = 0;
        if (part.equals("1")) {
            movex = 3;
            movey = 5;
        } else if (part.equals("2")) {
            movex = 1;
            movey = 3;
        }
        move(movex);
        turnRight();
        move(movey);
        turnLeft();
    }
}
